#include<bits/stdc++.h>
using namespace std;
void isPalindrome(string str) 
{ 
	int l = 0; 
	int h = str.length() - 1; 

	while (h > l) 
	{ 
		if (str[l++] != str[h--]) 
		{ 
			printf("NO\n"); 
			return; 
		} 
	} 
	printf("YES\n");
} 

int main() 
{ 
    int t;
    cin>>t;
    while(t--){
        string str;
        cin>>str;
        isPalindrome(str);
    }

	return 0; 
}
